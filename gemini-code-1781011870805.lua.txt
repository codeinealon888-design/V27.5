-- 3. ระบบฟังก์ชัน God Mode (ล็อคสถานะเลือดไม่ให้ลดลง)
function Modules:ToggleGodMode(state)
    local hum = LocalPlayer.Character and LocalPlayer.Character:FindFirstChildOfClass("Humanoid")
    if state then
        Omega.GodConn = hum and hum.HealthChanged:Connect(function(health) 
            if health < hum.MaxHealth then hum.Health = hum.MaxHealth end 
        end)
        if hum then hum.Health = hum.MaxHealth end
        Modules:LogMessage("God Mode: ENABLED")
    else
        if Omega.GodConn then Omega.GodConn:Disconnect(); Omega.GodConn = nil end
        Modules:LogMessage("God Mode: DISABLED")
    end
end

-- 4. ระบบฟังก์ชัน Chat Spy (ดักฟังข้อความกระซิบและการคุยในเซิร์ฟเวอร์)
local ChatConn1, ChatConn2
function Modules:ToggleChatSpy(state)
    Omega.ChatSpyActive = state
    if ChatConn1 then ChatConn1:Disconnect(); ChatConn1 = nil end
    if ChatConn2 then ChatConn2:Disconnect(); ChatConn2 = nil end
    
    if not state then 
        Modules:LogMessage("Chat Spy: DISABLED")
        return 
    end
    
    Modules:LogMessage("Chat Spy: ENABLED (Listening...)")
    
    local chatEvents = game:GetService("ReplicatedStorage"):FindFirstChild("DefaultChatSystemChatEvents")
    if chatEvents and chatEvents:FindFirstChild("OnMessageDoneFiltering") then
        ChatConn1 = chatEvents.OnMessageDoneFiltering.OnClientEvent:Connect(function(messageData)
            if Omega.ChatSpyActive and messageData and messageData.FromSpeaker and messageData.Message then
                if messageData.FromSpeaker ~= LocalPlayer.Name then
                    Modules:LogMessage("[SPY] [" .. messageData.FromSpeaker .. "]: " .. messageData.Message)
                end
            end
        end)
    end
    
    local textChatService = game:GetService("TextChatService")
    if textChatService and textChatService.ChatVersion == Enum.ChatVersion.TextChatService then
        ChatConn2 = textChatService.MessageReceived:Connect(function(textChatMessage)
            if Omega.ChatSpyActive and textChatMessage.TextSource and textChatMessage.Text then
                if textChatMessage.TextSource.Name ~= LocalPlayer.Name then
                    Modules:LogMessage("[SPY] [" .. textChatMessage.TextSource.Name .. "]: " .. textChatMessage.Text)
                end
            end
        end)
    end
end

-- 5. ระบบฟังก์ชัน Admin Detect & Instant Leave (สแกนผู้พัฒนาและสลับห้องหนี)
local AdminConn
local function CheckForAdmin(player)
    if not player or player == LocalPlayer then return end
    local isCreator = (player.UserId == game.CreatorId and game.CreatorId ~= 0)
    local isRobloxStaff = false
    pcall(function() isRobloxStaff = player:IsInGroup(1200769) end)
    
    if isCreator or isRobloxStaff then
        Modules:LogMessage("⚠️ ADMIN DETECTED -> " .. player.Name)
        task.wait(0.3)
        pcall(function()
            local servers = HttpService:JSONDecode(game:HttpGet("https://games.roblox.com/v1/games/" .. game.PlaceId .. "/servers/Public?sortOrder=Asc&limit=100"))
            for _, server in pairs(servers.data) do
                if server.playing < server.maxPlayers and server.id ~= game.JobId then
                    TeleportService:TeleportToPlaceInstance(game.PlaceId, server.id, LocalPlayer)
                    break
                end
            end
        end)
        task.wait(2)
        LocalPlayer:Kick("🛡️ Admin Detected!")
    end
end

function Modules:ToggleAdminDetect(state)
    Omega.AdminDetectActive = state
    if AdminConn then AdminConn:Disconnect(); AdminConn = nil end
    if not state then Modules:LogMessage("Admin Detect: DISABLED") return end
    
    Modules:LogMessage("Admin Detect: ENABLED (Scanning...)")
    for _, p in pairs(Players:GetPlayers()) do CheckForAdmin(p) end
    AdminConn = Players.PlayerAdded:Connect(function(p) if Omega.AdminDetectActive then CheckForAdmin(p) end end)
end
